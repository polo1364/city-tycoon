// 購買決策評分
function evaluatePurchase(player, tile) {
  let score = tile.price / 1000; // 基礎分
  
  // 壟斷潛力
  if (canCompleteMonopoly(player, tile)) {
    score += 50 * player.aiPersonality.monopolyFactor;
  }
  
  // 現金流安全
  let cashAfter = player.cash - tile.price;
  if (cashAfter < player.cash * player.aiPersonality.riskTolerance) {
    return 0; // 拒絕購買
  }
  
  return score;
}

5. 房屋升級

// 檢查是否可以建造
function canBuildHouse(player, tileIdx) {
  let tile = tiles[tileIdx];
  
  // 必須擁有完整同色組
  if (!hasMonopoly(player, tile.group)) return false;
  
  // 必須平均建造
  let groupTiles = getGroupTiles(tile.group);
  let currentLevel = houseCount(tileIdx);
  let maxLevel = Math.max(...groupTiles.map(t => houseCount(t)));
  
  if (currentLevel < maxLevel - 1) return false;
  
  return true;
}

進階技巧

1. 地產投資策略

•
優先購買同色組地產，形成壟斷

•
橘色組（日月潭、阿里山、溪頭）性價比最高

•
車站收益穩定，建議早期購買

•
深藍組（澎湖、馬祖）價格高但回報率低

2. 房屋升級策略

•
先將同色組升級到2棟，再考慮更高等級

•
優先升級對手可能踩到的地產

•
保留足夠現金應對租金支付

•
旅館回報率最高，但投資巨大

3. 卡牌使用技巧

•
保護卡在關鍵時刻使用（避免入獄或高額租金）

•
傳送卡可用於避開對手的高級地產

•
雙倍收租卡在對手即將踩到你的旅館時使用

•
交換位置卡可讓對手踩到你的陷阱

4. AI 對抗策略

•
激進型 AI：阻止其完成壟斷

•
保守型 AI：積極擴張，搶佔先機

•
平衡型 AI：穩健對抗，避免冒險

•
投機型 AI：利用其現金流不足的弱點
